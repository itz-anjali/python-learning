shutil  Personal Assistant hai jo files ko copy, move, aur delete
 karne ka heavy lifting karta hai.

~ * copy() and copy2() *

 # Case 1: Sirf content copy karna hai
shutil.copy('homework.txt', 'homework_backup.txt') 

# Case 2: Content + Date/Time sab same rakhna hai (Best for backups)
shutil.copy2('secret_plan.txt', 'secret_plan_final.txt')

~  * copytree *

shutil.copytree recursively
 (folder ke andar folder) sab kuch copy kar deta hai

 # Poora 'Project_A' folder utha ke 'Project_B' bana diya
shutil.copytree('C:/Users/Project_A', 'D:/Backups/Project_B')

~ * move() *

# File ko downloads se documents me bhejna
shutil.move('C:/Downloads/song.mp3', 'D:/Music/song.mp3')

~ * rmtree() *

# WARNING: Ye wapas nahi aayega!
shutil.rmtree('old_useless_folder')

shutil.disk_usage() (Space Check Karna)

# 'my_photos' folder ko 'holiday_pics.zip' bana dega
shutil.make_archive('holiday_pics', 'zip', 'my_photos')
